RETRY_COUNT=900
RETRY_INTERVAL=10

is_app_created_once=false
for i in $(seq 1 $RETRY_COUNT);
do
    echo -n "Application Status Check for iteration $i ";

    app_state_all=`${SPARK_HOME}/bin/spark-submit   --master $KUBE_CONTROL_PLANE --conf "spark.kubernetes.trust.certificates=true" --status $KUBE_NS:$DRIVER_POD_NAME --conf "spark.kubernetes.submission.connectionTimeout=60000" --conf "spark.kubernetes.submission.requestTimeout=60000" --conf "spark.kubernetes.driver.connectionTimeout=60000" --conf "spark.kubernetes.driver.requestTimeout=60000"`

    echo $app_state_all

    /usr/local/bin/spark_k8s_app_state_detector.sh  $KUBE_CONTROL_PLANE $KUBE_NS $DRIVER_POD_NAME
    m_exit_code=$?
    if [[ $m_exit_code -eq 255 ]]; then # App is running
        is_app_created_once=true
        sleep $RETRY_INTERVAL;
    elif [[ $m_exit_code -eq 254 ]]; then # App could not found
        if [ "$is_app_created_once" = true ] ; then
            echo "Pod is not reachable. Failing script."
            exit 1
        else
            sleep $RETRY_INTERVAL;
        fi
    elif [[ $m_exit_code -eq 0 ]]; then
        oc delete pod $DRIVER_POD_NAME -n spark
        echo "Manually detected exit code is: $m_exit_code"
        echo "Deleting pod [$DRIVER_POD_NAME]."
        exit $m_exit_code
    else
        #oc delete pod $DRIVER_POD_NAME -n spark
        echo "Manually detected exit code is: $m_exit_code"
       exit $m_exit_code
    fi
done
